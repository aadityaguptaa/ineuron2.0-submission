SELECT CONCAT(P.FirstName, ' ', P.LastName) AS Name, EE.JobTitle As JobTitle, EE.OrganizationLevel AS OrganizationalLevel, (SELECT COUNT(*) FROM HumanResources.Employee EEE WHERE EEE.OrganizationLevel = EE.BusinessEntityID) as NumberOfEmployeeReporting FROM HumanResources.Employee EE, Person.Person P WHERE EE.BusinessEntityID = P.BusinessEntityID AND EE.BusinessEntityID IN (SELECT DISTINCT OrganizationLevel FROM HumanResources.Employee E);
